// Exercício, uma função para apresentar uma pessoa, usando um objeto como fonte de dados.
function apresenta(pessoa) {
    let trataG = "uma", trataI = "jovem", trataA = "a";

    if (pessoa.genero === "feminino") {
        trataG = "uma";
        trataA = "a"
    } else {
        trataG = "um";
        trataA = "o";
    }

    if (pessoa.idade < 25) {
        trataI = "jovem";
    } else if (pessoa.idade < 65) {
        trataI = "adult" + trataA;
    } else {
        trataI = "idos" + trataA;
    }

    console.log(`Olá! Meu nome é ${pessoa.nome}, sou ${trataG} ${trataI} de ${pessoa.idade} anos, ${pessoa.altura} de altura e sou ${pessoa.profissao}.`);
}


const pessoa = {
    nome: "Giuliana",
    genero: "feminino",
    idade: 41,
    profissao: "Programadora",
    altura: 1.74
}

apresenta(pessoa);

// Exercício - Refatorado, funções em objetos.

const pessoa = {
    nome: "Giuliana",
    genero: "feminino",
    idade: 41,
    profissao: "Programadora",
    altura: 1.74,
    apresentar: function () {
        let trataG = "uma", trataI = this.faixaEtaria(this.idade), trataA = "a";

        if (this.genero === "feminino") {
            trataG = "uma";
            trataA = "a"
        } else {
            trataG = "um";
            trataA = "o";
        }

        console.log(`Olá! Meu nome é ${this.nome}, sou ${trataG} ${trataI}${trataA} de ${this.idade} anos, ${this.altura} de altura e sou ${this.profissao}.`);
    },
    faixaEtaria: function () {
        let trataI = "";
        if (this.idade < 25) {
            trataI = "jovem";
        } else if (this.idade < 65) {
            trataI = "adult";
        } else {
            trataI = "idos";
        }
        return trataI;
    }
}

pessoa.apresentar();
console.log(pessoa.faixaEtaria());
